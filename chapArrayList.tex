\section{Get the Middle node in Single Linked List}

\begin{quote}
\begin{verbatim}
/**
 * node struct
 * struct Node
 * {
 *  Node* next;
 *  int val;
 *  Node(_int val): val(val),next(0){}
 * };
 */

// using two pinters, fast and slow pointers to get the middle node
Node* middle(Node* head)
{
  if ( !head ||!head->next )
      return p;
  
  Node * slow = head;
  Node * fast = head;
  while(fast->next && fast->next->next)
  {
    fast = fast->next->next;
    slow = slow->next;  
  }
  return slow;
}

\end{verbatim}
\end{quote}

void sortColors(int A[], int n) 
{
	//red is begin from beginning, blue begin from end
	int red = 0, blue = n - 1;
	for(int i=0;i<n;++i)
	{
		if(A[i]==0)
		{
			 swap(A[i],A[red]);
			 red++;
		}
		else if(A[i]==2)
		{
			swap(A[i],A[blue]);
			 blue--;
		}
	}
}
